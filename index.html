async function runSimulation() { const netlist = document.getElementById('netlistInput').value.trim(); const outputArea = document.getElementById('outputArea'); const runBtn = document.getElementById('runBtn'); const btnLabel = document.getElementById('btnLabel'); const statusBar = document.getElementById('statusBar'); const outputBadge = document.getElementById('outputBadge'); if (!netlist) { outputArea.innerHTML = '⚠ ERROR: No netlist provided.'; return; } runBtn.classList.add('loading'); btnLabel.innerHTML = ' SIMULATING...'; statusBar.textContent = 'RUNNING...'; outputBadge.textContent = 'PROCESSING'; const t0 = performance.now(); outputArea.innerHTML = '⟳ Dispatching netlist to ngspice engine...'; try { const response = await fetch('https://final-spice-as-a-service.onrender.com/simulate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ netlist }) }); const data = await response.json(); const elapsed = ((performance.now() - t0) / 1000).toFixed(3); if (response.ok) { outputArea.innerHTML = parseAndRender(data.output, elapsed); statusBar.textContent = `DONE IN ${elapsed}s`; outputBadge.textContent = `SUCCESS · ${elapsed}s`; } else { outputArea.innerHTML = `✖ ERROR ${response.status}\n${esc(data.detail)}`; statusBar.textContent = 'FAILED'; outputBadge.textContent = 'ERROR'; } } catch (err) { outputArea.innerHTML = `✖ CONNECTION REFUSED\n\nCould not reach the simulation server.\nMake sure the backend is running.`; statusBar.textContent = 'NO CONNECTION'; outputBadge.textContent = 'OFFLINE'; } finally { runBtn.classList.remove('loading'); btnLabel.innerHTML = ' RUN SIMULATION'; } } // ─── STRICT SMART PARSER ────────────────────────────────────────── function parseAndRender(raw, elapsed) { const lines = raw.split('\n'); const warnings = []; const errors = []; const opRows = []; const printRows = []; let colNames = []; let headerLine = null; let inOP = false; // 1. Collect warnings and errors (Ignoring the 'exp' trap) lines.forEach(l => { if (/^\s*warning/i.test(l) && l.trim().length > 8) warnings.push(l.trim()); if (/^\s*error/i.test(l) && l.trim().length > 5 && !/warning/i.test(l)) errors.push(l.trim()); }); // 2. Parse Data Line by Line for (let i = 0; i < lines.length; i++) { const l = lines[i].trim(); if (!l) continue; // --- Operating Point (.op) Parsing --- if (/operating point/i.test(l)) { inOP = true; continue; } if (inOP && /^[A-Z][a-z]+ (Analysis|Characteristic|Sweep)/.test(l)) { inOP = false; } if (inOP) { const opMatch = l.match(/^([a-zA-Z0-9_#()]+)\s*(?:=)?\s*([-\d.eE+]+)$/); if (opMatch) { const name = opMatch[1].toLowerCase(); const valStr = opMatch[2]; if (!['node', 'voltage', 'source', 'current', 'device', 'model'].includes(name)) { let unit = ''; let note = 'Parameter'; if (name.endsWith('#branch') || ['id','ic','ib','ie'].includes(name) || name.startsWith('i(')) { unit = 'A'; note = 'Current'; } else if (['vgs','vds','vth','vdsat','vbs','vbe','vce'].includes(name) || name.startsWith('v(')) { unit = 'V'; note = 'Voltage'; } else if (['gm','gds'].includes(name)) { unit = 'S'; note = 'Conductance'; } else if (['cbd','cbs','cgs','cgd'].includes(name)) { unit = 'F'; note = 'Capacitance'; } else { unit = 'V'; note = 'Node Voltage'; } opRows.push({ name: name.toUpperCase(), value: formatVal(valStr), unit: unit, note: note }); } } continue; } // --- Tabular Data (.print) Parsing --- if (!headerLine && /^(index|freq|time)\s+/i.test(l)) { headerLine = l; colNames = l.split(/\s+/).filter(Boolean); continue; } // FIXED: Must strictly start with a digit (Index number), followed by a space, then data. if (headerLine && /^\d+\s+[-\d.eE+]+/.test(l)) { const vals = l.split(/\s+/).filter(Boolean); // Double check it's an actual data row matching column count if (vals.length >= 2) { printRows.push(vals); } } if (headerLine && printRows.length > 0 && (l.includes('---') || l.startsWith('Y') || /^[a-zA-Z]/.test(l))) { headerLine = null; } } // 3. Build HTML Output let html = '
'; if (errors.length) { html += `
SIMULATION FAILED
${esc(errors[0])}
`; } if (warnings.length) { html += `
ENGINE WARNING
${esc(warnings[0])}
`; } if (opRows.length) { html += `
⚡ DC OPERATING POINT
${opRows.map(r => `
${esc(r.name)}
${esc(r.value)}
${esc(r.unit)}
${esc(r.note)}
`).join('')}
`; } if (printRows.length && colNames.length) { const maxRows = 25; const displayRows = printRows.slice(0, maxRows); html += `
📊 SWEEP / TRANSIENT DATA
${colNames.map(c => ``).join('')} ${displayRows.map(row => ` ${row.map((v, i) => ``).join('')} `).join('')}
${esc(c)}
${formatSci(v)}
`; if (printRows.length > maxRows) { html += `
... Showing first ${maxRows} of ${printRows.length} data points. Check raw output for full dataset.
`; } } let rawDisplayStyle = "display:none;"; if (!opRows.length && !printRows.length && !errors.length && !warnings.length) { html += `
NO STRUCTURED DATA FOUND
Ngspice ran successfully, but no matching table data was generated. See raw output below.
`; rawDisplayStyle = "display:block;"; } html += `
⏱ ${elapsed}s ${warnings.length ? `${warnings.length} WARN` : ''} ${errors.length ? `${errors.length} ERR` : ''}
${esc(cleanRaw(raw))}
`; return html; } // ─── UTILITY FUNCTIONS ────────────────────────────────────────── function toggleRaw(btn) { const raw = btn.nextElementSibling; const visible = raw.style.display === 'block'; raw.style.display = visible ? 'none' : 'block'; } function formatVal(s) { const n = parseFloat(s); if (isNaN(n)) return s; const abs = Math.abs(n); if (abs === 0) return '0'; if (abs >= 1e9) return (n/1e9).toPrecision(4) + ' G'; if (abs >= 1e6) return (n/1e6).toPrecision(4) + ' M'; if (abs >= 1e3) return (n/1e3).toPrecision(4) + ' k'; if (abs >= 1) return n.toPrecision(5); if (abs >= 1e-3) return (n*1e3).toPrecision(4) + ' m'; if (abs >= 1e-6) return (n*1e6).toPrecision(4) + ' μ'; if (abs >= 1e-9) return (n*1e9).toPrecision(4) + ' n'; if (abs >= 1e-12)return (n*1e12).toPrecision(4) + ' p'; return n.toExponential(3); } function formatSci(s) { const n = parseFloat(s); return isNaN(n) ? esc(s) : formatVal(s); } function cleanRaw(raw) { return raw.split('\n') .filter(l => !/^(ngspice|note:|circuit:|date:|temperature:|^\s*$|^-{5,}|tnom|copyright|license|please|for further)/i.test(l)) .join('\n').trim(); } function clearAll() { document.getElementById('netlistInput').value = ''; document.getElementById('outputArea').innerHTML = '── Cleared. Enter a netlist and run. █'; document.getElementById('statusBar').textContent = 'READY'; document.getElementById('outputBadge').textContent = 'AWAITING INPUT'; } function loadExample() { document.getElementById('netlistInput').value = `RC Low-Pass Filter AC Sweep Vin in 0 AC 1V R1 in out 1k C1 out 0 1u .ac dec 100 1 100k .print ac v(out) vp(out) .end`; } function esc(s) { return String(s).replace(/&/g,'&').replace(//g,'>'); }
